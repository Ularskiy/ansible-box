# Структура репозитория ansible-box

```text
ansible-box/
├── ansible.cfg                  # Конфигурация Ansible (инвентарь, роли, пользователь)
├── inventory/                   # Инвентарь хостов (разделён по окружениям)
│   ├── production/
│   │   └── hosts.ini
│   └── staging/
│       └── hosts.ini
├── group_vars/                  # Переменные для групп хостов
│   ├── all.yml                  # Общие переменные (user, timezone и т.д.)
│   └── <group>.yml              # Например: web.yml, db.yml
├── host_vars/                   # Переменные для конкретных хостов
│   └── <hostname>.yml           # Например: web01.yml, db01.yml
├── playbooks/                   # Плейбуки (сценарии запуска)
│   ├──                          # Основной плейбук, вызывает всё остальное
│   ├──                          # Настройка ОС, SSH, сеть
│   ├──                          # Развёртывание БД
│   ├──                          # Развёртывание приложения
│   └──                          # Установка мониторинга (Prometheus, Grafana)
├── roles/                       # Ansible-роли (по логике)
│   ├── infra/                   # Инфраструктурные роли (сеть, ssh, пакеты)
│   │   └── 
│   ├── deploy/                  # Роли деплоя приложений
│   │   └── 
│   └── shared/                  # Общие роли (пользователи, timezone и т.д.)
│       └── 
├── templates/                   # Jinja2-шаблоны конфигураций
│   └── 
├── files/                       # Статичные файлы (копируются как есть)
│   └── 
├── library/                     # Кастомные модули Ansible на Python
│   └── 
├── LICENSE                      # Лицензия проекта
└── README.md                    # Главное описание проекта
```
